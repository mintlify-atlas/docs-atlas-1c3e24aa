## Method Signature

```typescript
export const deleteClient = async (clientId: string): Promise<{ success: boolean }>
```

<Note>

</Note>

## Description

Permanently removes a client record from the database using the client's unique identifier. This operation cannot be undone.

<Warning>
  This is a destructive operation. Always confirm with the user before deleting a client. Consider checking for associated projects before deletion.
</Warning>

## Parameters

<ParamField path="clientId" type="string" required>
  Unique identifier of the client to delete
</ParamField>

## Return Type

Returns a Promise that resolves to an object indicating the success of the operation.

<ResponseField name="response" type="object">
  Deletion result object
  
  <Expandable title="Response properties">
    <ResponseField name="success" type="boolean">
      Indicates whether the deletion was successful. `true` if the client was deleted, `false` otherwise.
    </ResponseField>
  </Expandable>
</ResponseField>

```typescript With Undo Functionality
import { deleteClient, addClient, getClients } from './services/api';
import type { Client } from './types';

interface DeleteAction {
  clientId: string;
  clientData: Client;
  timestamp: number;
}

class ClientDeleteManager {
  private deleteHistory: DeleteAction[] = [];
  private readonly UNDO_TIMEOUT = 5000; // 5 seconds

  async deleteWithUndo(client: Client): Promise<boolean> {
    try {
      const result = await deleteClient(client.id);
      
      if (result.success) {
        // Store for potential undo
        this.deleteHistory.push({
          clientId: client.id,
          clientData: client,
          timestamp: Date.now()
        });
        
        console.log('Client deleted. You have 5 seconds to undo.');
        
        // Auto-expire undo option
        setTimeout(() => {
          this.deleteHistory = this.deleteHistory.filter(
            action => Date.now() - action.timestamp < this.UNDO_TIMEOUT
          );
        }, this.UNDO_TIMEOUT);
      }
      
      return result.success;
    } catch (error) {
      console.error('Delete failed:', error);
      throw error;
    }
  }

  async undo(clientId: string): Promise<boolean> {
    const actionIndex = this.deleteHistory.findIndex(a => a.clientId === clientId);
    
    if (actionIndex === -1) {
      console.log('No recent delete action found for this client');
      return false;
    }
    
    const action = this.deleteHistory[actionIndex];
    
    // Check if still within undo window
    if (Date.now() - action.timestamp > this.UNDO_TIMEOUT) {
      console.log('Undo window expired');
      return false;
    }
    
    try {
      // Recreate the client
      const { id, ...clientData } = action.clientData;
      await addClient(clientData);
      
      // Remove from history
      this.deleteHistory.splice(actionIndex, 1);
      
      console.log('Delete undone successfully');
      return true;
    } catch (error) {
      console.error('Undo failed:', error);
      return false;
    }
  }
}
```

## Error Handling

<Warning>
  Always verify that the client has no dependencies (like projects) before deletion to maintain data integrity.
</Warning>

### Common Errors

<AccordionGroup>
  <Accordion title="User is not authenticated. No session found.">
    **Cause**: No valid session found in localStorage
    
    **Solution**: Ensure the user is logged in before allowing deletions
  </Accordion>
  
  <Accordion title="Client not found">
    **Cause**: No client exists with the provided ID
    
    **Solution**: Verify the client exists before attempting deletion:
    
    ```typescript
    import { getClients, deleteClient } from './services/api';
    
    const deleteClientIfExists = async (clientId: string) => {
      const clients = await getClients();
      const exists = clients.some(c => c.id === clientId);
      
      if (!exists) {
        throw new Error('Client not found. It may have already been deleted.');
      }
      
      return await deleteClient(clientId);
    };
    ```
  </Accordion>
  
  <Accordion title="Cannot delete client with associated projects">
    **Cause**: The client has related projects that would be orphaned
    
    **Solution**: Delete or reassign all associated projects first:
    
    ```typescript
    import { deleteClient, getProjects, deleteProject } from './services/api';
    
    const deleteClientAndProjects = async (clientId: string) => {
      // Get and delete all client projects
      const projects = await getProjects();
      const clientProjects = projects.filter(p => p.clientId === clientId);
      
      for (const project of clientProjects) {
        await deleteProject(project.id);
      }
      
      // Now delete the client
      return await deleteClient(clientId);
    };
    ```
  </Accordion>
  
  <Accordion title="Failed to fetch">
    **Cause**: Cannot connect to the Google Apps Script backend
    
    **Solution**: Verify the `DATA_SCRIPT_URL` configuration and deployment settings
  </Accordion>
</AccordionGroup>

## Best Practices

<Steps>
  <Step title="Always confirm with user">
    Show a confirmation dialog before deleting to prevent accidental deletions
  </Step>
  
  <Step title="Check for dependencies">
    Verify the client has no associated projects or other related data
  </Step>
  
  <Step title="Provide clear feedback">
    Show success or error messages to keep users informed
  </Step>
  
  <Step title="Consider soft delete">
    Instead of permanent deletion, consider marking records as inactive
  </Step>
  
  <Step title="Update UI immediately">
    Remove the deleted client from the UI right after successful deletion
  </Step>
  
  <Step title="Implement undo">
    Provide a brief window for users to undo accidental deletions
  </Step>
</Steps>

<Note>
  **Data Integrity**: Always check for related records (projects, notes, etc.) before deleting a client to maintain referential integrity in your application.
</Note>

## Security Considerations

- Only authenticated users with valid sessions can delete clients
- The backend should verify the user has permission to delete from the specified spreadsheet
- Consider implementing role-based access control for delete operations

## Implementation Details

The method internally calls:

```typescript
return callAppsScript('deleteClient', clientId);
```

This sends a POST request to the Google Apps Script backend with:
- Action: `'deleteClient'`
- Payload: Client ID + spreadsheet ID from session
- Returns: Object with `success` boolean
