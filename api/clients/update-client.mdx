## Description

Updates an existing client record in the database. The method requires the complete `Client` object including the `id` field to identify which client to update.

```typescript Partial Update
import { getClients, updateClient } from './services/api';
import type { Client } from './types';

// Update only specific fields while preserving others
const updateClientPartial = async (
  clientId: string,
  updates: Partial<Omit<Client, 'id'>>
) => {
  try {
    // First, fetch the current client data
    const clients = await getClients();
    const existingClient = clients.find(c => c.id === clientId);
    
    if (!existingClient) {
      throw new Error(`Client with ID ${clientId} not found`);
    }
    
    // Merge existing data with updates
    const updatedClient = await updateClient({
      ...existingClient,
      ...updates
    });
    
    return updatedClient;
  } catch (error) {
    console.error('Failed to update client:', error);
    throw error;
  }
};

// Usage
await updateClientPartial('client_123', {
  email: 'newemail@example.com',
  phone: '+9999999999'
});
```

```typescript React Edit Form
import { updateClient } from './services/api';
```

```typescript Batch Update
import { updateClient } from './services/api';
import type { Client } from './types';

interface BatchUpdateResult {
  success: Client[];
  failed: Array<{ client: Client; error: string }>;
}

const updateMultipleClients = async (
  clients: Client[]
): Promise<BatchUpdateResult> => {
  const result: BatchUpdateResult = {
    success: [],
    failed: []
  };
  
  for (const client of clients) {
    try {
      const updated = await updateClient(client);
      result.success.push(updated);
      console.log(`✓ Updated: ${client.name}`);
    } catch (error) {
      const errorMessage = error instanceof Error ? error.message : 'Unknown error';
      result.failed.push({ client, error: errorMessage });
      console.error(`✗ Failed: ${client.name} - ${errorMessage}`);
    }
  }
  
  console.log(`Batch update: ${result.success.length} succeeded, ${result.failed.length} failed`);
  return result;
};
```

## Error Handling

<Warning>
  Always ensure the client exists before attempting to update. Invalid IDs will result in backend errors.
</Warning>

### Common Errors

<AccordionGroup>
  <Accordion title="User is not authenticated. No session found.">
    **Cause**: No valid session found in localStorage
    
    **Solution**: Ensure the user is logged in before allowing updates
  </Accordion>
  
  <Accordion title="Client not found">
    **Cause**: No client exists with the provided ID
    
    **Solution**: Verify the client ID is correct and the client hasn't been deleted:
    
    ```typescript
    import { getClients, updateClient } from './services/api';
    
    const updateClientSafely = async (clientData: Client) => {
      const clients = await getClients();
      const exists = clients.some(c => c.id === clientData.id);
      
      if (!exists) {
        throw new Error('Client not found. It may have been deleted.');
      }
      
      return await updateClient(clientData);
    };
    ```
  </Accordion>
  
  <Accordion title="Phone number already exists">
    **Cause**: Another client already uses the new phone number
    
    **Solution**: Validate phone uniqueness before updating:
    
    ```typescript
    import { getClientByPhone, updateClient } from './services/api';
    
    const updateClientPhone = async (client: Client, newPhone: string) => {
      if (client.phone !== newPhone) {
        const existing = await getClientByPhone(newPhone);
        if (existing && existing.id !== client.id) {
          throw new Error('Phone number already in use by another client');
        }
      }
      
      return await updateClient({ ...client, phone: newPhone });
    };
    ```
  </Accordion>
  
  <Accordion title="Invalid data format">
    **Cause**: Backend validation failed
    
    **Solution**: Ensure all fields are properly formatted and required fields are not empty
  </Accordion>
</AccordionGroup>

## Best Practices

<Steps>
  <Step title="Fetch current data first">
    Always retrieve the latest client data before updating to avoid overwriting concurrent changes
  </Step>
  
  <Step title="Validate changes">
    Verify that the new data is valid and different from the current data
  </Step>
  
  <Step title="Handle conflicts">
    Implement conflict resolution for concurrent edits
  </Step>
  
  <Step title="Provide feedback">
    Give users clear success/error messages after the update
  </Step>
  
  <Step title="Optimistic updates">
    Update the UI immediately for better UX, but be ready to rollback on errors
  </Step>
</Steps>

<Note>
  **Immutable Updates**: Always create a new object when updating rather than mutating the existing client object.
</Note>

## Implementation Details

The method internally calls:

```typescript
return callAppsScript('updateClient', clientData);
```

This sends a POST request to the Google Apps Script backend with:
- Action: `'updateClient'`
- Payload: Complete client object + spreadsheet ID from session
- Returns: Updated client data
