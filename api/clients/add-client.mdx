## Method Signature

<Note>

## Description

Creates a new client record in the database. The method accepts client data without an `id` field, as the ID is automatically generated by the backend.

```typescript With Validation
import { addClient } from './services/api';

interface ClientInput {
  name: string;
  contactPerson: string;
  email: string;
  phone: string;
}

function validateClientData(data: ClientInput): string[] {
  const errors: string[] = [];
  
  if (!data.name.trim()) {
    errors.push('Company name is required');
  }
  
  if (!data.contactPerson.trim()) {
    errors.push('Contact person is required');
  }
  
  const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
  if (!emailRegex.test(data.email)) {
    errors.push('Invalid email format');
  }
  
  const phoneRegex = /^\+?[1-9]\d{1,14}$/;
  if (!phoneRegex.test(data.phone.replace(/[\s-]/g, ''))) {
    errors.push('Invalid phone number format');
  }
  
  return errors;
}

const createClientWithValidation = async (data: ClientInput) => {
  // Validate input
  const errors = validateClientData(data);
  if (errors.length > 0) {
    throw new Error(`Validation failed: ${errors.join(', ')}`);
  }
  
  try {
    const newClient = await addClient(data);
    return newClient;
  } catch (error) {
    console.error('Failed to create client:', error);
    throw error;
  }
};
```

```typescript Bulk Import
import { addClient } from './services/api';
import type { Client } from './types';

interface ImportResult {
  success: Client[];
  failed: Array<{ data: Omit<Client, 'id'>; error: string }>;
}

const importClients = async (
  clientsData: Array<Omit<Client, 'id'>>
): Promise<ImportResult> => {
  const result: ImportResult = {
    success: [],
    failed: []
  };
  
  for (const clientData of clientsData) {
    try {
      const newClient = await addClient(clientData);
      result.success.push(newClient);
      console.log(`✓ Created: ${clientData.name}`);
    } catch (error) {
      const errorMessage = error instanceof Error ? error.message : 'Unknown error';
      result.failed.push({ data: clientData, error: errorMessage });
      console.error(`✗ Failed: ${clientData.name} - ${errorMessage}`);
    }
  }
  
  console.log(`Import complete: ${result.success.length} succeeded, ${result.failed.length} failed`);
  return result;
};
```

## Error Handling

<Warning>
  Always validate client data before calling `addClient()` to ensure data integrity.
</Warning>

### Common Errors

<AccordionGroup>
  <Accordion title="User is not authenticated. No session found.">
    **Cause**: No valid session found in localStorage
    
    **Solution**: Ensure the user is logged in before allowing client creation
  </Accordion>
  
  <Accordion title="Duplicate phone number">
    **Cause**: A client with the same phone number already exists
    
    **Solution**: Check for existing clients before adding:
    
    ```typescript
    import { getClientByPhone, addClient } from './services/api';
    
    const addClientSafely = async (clientData: Omit<Client, 'id'>) => {
      // Check if client exists
      const existing = await getClientByPhone(clientData.phone);
      if (existing) {
        throw new Error('A client with this phone number already exists');
      }
      
      return await addClient(clientData);
    };
    ```
  </Accordion>
  
  <Accordion title="Invalid data format">
    **Cause**: Backend validation failed
    
    **Solution**: Ensure all required fields are provided and properly formatted
  </Accordion>
  
  <Accordion title="Failed to fetch">
    **Cause**: Cannot connect to the Google Apps Script backend
    
    **Solution**: Verify the `DATA_SCRIPT_URL` configuration and deployment settings
  </Accordion>
</AccordionGroup>

## Implementation Details

The method internally calls:

```typescript
return callAppsScript('addClient', clientData);
```

This sends a POST request to the Google Apps Script backend with:
- Action: `'addClient'`
- Payload: Client data object + spreadsheet ID from session
- Returns: Newly created client with generated ID
