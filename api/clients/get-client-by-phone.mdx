## Description

Searches for and retrieves a client record using their phone number as the lookup key. This method is useful for checking if a client already exists before creating a new record, or for quickly finding a client by their contact information.

## Parameters

<ParamField path="phone" type="string" required>
  The phone number to search for. Should match the format stored in the database.
</ParamField>

<Note>
  Phone number format matters. Ensure the search string matches the format used when the client was created (e.g., with or without country code, spaces, or dashes).
</Note>

## Return Type

Returns a Promise that resolves to either a `Client` object if found, or `null` if no client exists with the specified phone number.

```typescript React Search Component
import { useState } from 'react';
import { getClientByPhone } from './services/api';

function ClientSearchByPhone() {

              <h3>{client.name}</h3>
```

```typescript Validation Before Update
const updateClientPhone = async (
  client: Client,
  newPhone: string
): Promise<Client> => {
  // Only check if phone number is actually changing
  if (client.phone === newPhone) {
    return client;
  }
  
  try {
    // Check if new phone number is already in use
    const existingClient = await getClientByPhone(newPhone);
    
    if (existingClient && existingClient.id !== client.id) {
      throw new Error(
        `Phone number ${newPhone} is already in use by ${existingClient.name}`
      );
    }
    
    // Safe to update
    const updatedClient = await updateClient({
      ...client,
      phone: newPhone
    });
    
    return updatedClient;
  } catch (error) {
    console.error('Failed to update phone:', error);
    throw error;
  }
};
```

```typescript Batch Phone Lookup
import { getClientByPhone } from './services/api';
import type { Client } from './types';

interface PhoneLookupResult {
  phone: string;
  client: Client | null;
}

const lookupMultiplePhones = async (
  phoneNumbers: string[]
): Promise<PhoneLookupResult[]> => {
  const results: PhoneLookupResult[] = [];
  
  for (const phone of phoneNumbers) {
    try {
      const client = await getClientByPhone(phone);
      results.push({ phone, client });
      
      if (client) {
        console.log(`✓ Found: ${phone} -> ${client.name}`);
      } else {
        console.log(`✗ Not found: ${phone}`);
      }
    } catch (error) {
      console.error(`Error looking up ${phone}:`, error);
      results.push({ phone, client: null });
    }
  }
  
  const found = results.filter(r => r.client !== null).length;
  console.log(`Lookup complete: ${found}/${phoneNumbers.length} found`);
  
  return results;
};

// Usage
```

### Recommended Format

```typescript
// Store phone numbers in E.164 format: +[country code][number]
// Example: +12025551234

const formatPhone = (input: string): string => {
  // Remove all non-digit characters
  const digits = input.replace(/\D/g, '');
  
  // Add country code if missing (assuming US: +1)
  if (digits.length === 10) {
    return `+1${digits}`;
  }
  
  if (digits.length === 11 && digits.startsWith('1')) {
    return `+${digits}`;
  }
  
  return `+${digits}`;
};
```

## Implementation Details

The method internally calls:

```typescript
return callAppsScript('getClientByPhone', phone);
```

This sends a POST request to the Google Apps Script backend with:
- Action: `'getClientByPhone'`
- Payload: Phone number string + spreadsheet ID from session
- Returns: Client object or `null`

## Performance Considerations

- This method queries the backend database, so it involves network latency
- For multiple lookups, consider caching results or using `getClients()` once and filtering locally
- Phone number matching is case-sensitive and format-sensitive
