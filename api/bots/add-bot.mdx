## Method Signature

```typescript
addBot(
  botData: Omit<Bot, 'id' | 'createdAt' | 'status' | 'updatedAt' | 'onlineSince' | 'lastOnlineDuration'>
): Promise<Bot>
```

Creates a new bot instance in GestorPro. The bot will be created with `status: "offline"` and can be connected to WhatsApp using the QR code method.

## Parameters

<ParamField body="botData" type="object" required>
  Bot configuration object with the following fields:
  
  <Expandable title="Properties" defaultOpen>
    <ParamField body="name" type="string" required>
      Display name for the bot (e.g., "Customer Support Bot")
    </ParamField>
    
    <ParamField body="description" type="string" required>
      Detailed description of the bot's purpose and functionality
    </ParamField>
    
    <ParamField body="systemInstruction" type="string" required>
      AI system prompt that defines the bot's behavior, personality, and response style. This is sent to the OpenAI assistant to guide conversations.
    </ParamField>
    
    <ParamField body="id_builderBot" type="string" required>
      BuilderBot instance ID obtained from your BuilderBot account
    </ParamField>
    
    <ParamField body="builderBotApiKey" type="string">
      Optional BuilderBot API key for authenticated API requests
    </ParamField>
  </Expandable>
</ParamField>

<Note>
The following fields are automatically generated and should NOT be included in `botData`:
- `id` - Generated by the backend
- `createdAt` - Set to current timestamp
- `status` - Initialized to `"offline"`
- `updatedAt` - Set to current timestamp
- `onlineSince` - Undefined until bot connects
- `lastOnlineDuration` - Undefined until first disconnect
</Note>

## Returns

<ResponseField name="Promise<Bot>" type="Promise">
  Returns a promise that resolves to the newly created Bot object with all fields populated
  
  <Expandable title="Bot">
    <ResponseField name="id" type="string">
      Unique identifier generated by GestorPro
    </ResponseField>
    
    <ResponseField name="name" type="string">
      The bot name provided in botData
    </ResponseField>
    
    <ResponseField name="description" type="string">
      The bot description provided in botData
    </ResponseField>
    
    <ResponseField name="systemInstruction" type="string">
      The system instruction provided in botData
    </ResponseField>
    
    <ResponseField name="createdAt" type="string">
      ISO 8601 timestamp of creation
    </ResponseField>
    
    <ResponseField name="id_builderBot" type="string">
      The BuilderBot ID provided in botData
    </ResponseField>
    
    <ResponseField name="status" type="string">
      Will be `"offline"` for newly created bots
    </ResponseField>
    
    <ResponseField name="updatedAt" type="string">
      ISO 8601 timestamp, initially same as createdAt
    </ResponseField>
    
    <ResponseField name="builderBotApiKey" type="string">
      The API key if provided in botData
    </ResponseField>
  </Expandable>
</ResponseField>

## Error Handling

<CodeGroup>
```typescript Validation
import { addBot } from './services/api';

async function createBotWithValidation(
  name: string,
  description: string,
  systemInstruction: string,
  id_builderBot: string,
  builderBotApiKey?: string
) {
  // Client-side validation
  if (!name.trim()) {
    throw new Error('Bot name is required');
  }
  
  if (name.length > 100) {
    throw new Error('Bot name must be 100 characters or less');
  }
  
  if (!description.trim()) {
    throw new Error('Bot description is required');
  }
  
  if (!systemInstruction.trim()) {
    throw new Error('System instruction is required');
  }
  
  if (systemInstruction.length < 50) {
    throw new Error('System instruction should be at least 50 characters for effective AI guidance');
  }
  
  if (!id_builderBot.trim()) {
    throw new Error('BuilderBot ID is required');
  }

  try {
    return await addBot({
      name: name.trim(),
      description: description.trim(),
      systemInstruction: systemInstruction.trim(),
      id_builderBot: id_builderBot.trim(),
      builderBotApiKey: builderBotApiKey?.trim()
    });
  } catch (error) {
    console.error('Backend error:', error);
    throw error;
  }
}
```

import { addBot, deleteBot } from './services/api';

async function createBotSafely(botData: any) {
  let createdBot = null;
  
  try {

    console.log('Bot created:', createdBot.id);

    await initializeExternalSystem(createdBot.id_builderBot);
    
    return createdBot;
  } catch (error) {
    // If external initialization fails, remove the bot

      console.log('Rolling back bot creation...');
      await deleteBot(createdBot.id);
    }
    throw error;
  }
}

## Next Steps After Creation

After creating a bot, you typically need to:

<Steps>
  <Step title="Get QR Code">
    Call `getBotQrCode(bot.id_builderBot)` to get the WhatsApp connection QR code
  </Step>
  <Step title="Connect to WhatsApp">
    Display the QR code for the user to scan with their WhatsApp
  </Step>
  <Step title="Monitor Status">
    Poll `getBots()` to monitor when the bot status changes to `"online"`
  </Step>
  <Step title="Configure Flows">
    Use `getBotFlows(bot.id_builderBot)` to view and configure conversation flows
  </Step>
</Steps>

```typescript Complete Bot Setup
import { addBot, getBotQrCode, connectBot } from './services/api';

async function setupNewBot() {
  // 1. Create bot
  const bot = await addBot({
    name: "My Bot",
    description: "A helpful assistant",
    systemInstruction: "You are a helpful assistant.",
    id_builderBot: "bb_123"
  });
  
  console.log('Bot created:', bot.id);
  
  // 2. Get QR code for WhatsApp connection
  const { qr } = await getBotQrCode(bot.id_builderBot);
  console.log('QR code:', qr);
  
  // 3. Display QR code to user (in your UI)
  // displayQrCode(qr);
  
  // 4. After QR is scanned, connect the bot
  await connectBot(bot.id);
  console.log('Bot connected!');
  
  return bot;
}
```
