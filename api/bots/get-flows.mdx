## Method Signature

```typescript
getBotFlows(builderBotId: string): Promise<Flow[]>
```

Retrieves all conversation flows configured for a specific BuilderBot instance. Flows define the bot's conversational logic, including keywords, responses, and AI integrations.

## Parameters

<ParamField path="builderBotId" type="string" required>
  The BuilderBot instance ID (not the GestorPro bot ID). This is the `id_builderBot` field from the Bot object.
</ParamField>

```typescript Find Flow by Keyword
import { getBotFlows } from './services/api';

async function findFlowByKeyword(
  builderBotId: string,
  keyword: string
): Promise<Flow | null> {
  const flows = await getBotFlows(builderBotId);
  
  const matchingFlow = flows.find(flow =>
    flow.keyword.some(k => k.toLowerCase() === keyword.toLowerCase())
  );
  
  return matchingFlow || null;
}

// Usage
```

```typescript React Component
import { useState, useEffect } from 'react';
import { getBotFlows } from './services/api';
import type { Flow } from './types';

function FlowList({ builderBotId }: { builderBotId: string }) {
  const [flows, setFlows] = useState<Flow[]>([]);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  useEffect(() => {
    async function fetchFlows() {
      try {
        const data = await getBotFlows(builderBotId);
        // Filter out deleted flows
        const activeFlows = data.filter(f => !f.deletedAt);
        setFlows(activeFlows);
      } catch (err) {
        setError(err instanceof Error ? err.message : 'Failed to load flows');
      } finally {
        setLoading(false);
      }
    }

    fetchFlows();
  }, [builderBotId]);

  if (loading) return <div>Loading flows...</div>;
  if (error) return <div>Error: {error}</div>;

  return (
    <div className="flow-list">
      <h2>Conversation Flows ({flows.length})</h2>
      {flows.map(flow => (
        <div key={flow.id} className="flow-card">
          <h3>{flow.label}</h3>
          <div className="flow-keywords">
            Keywords: {flow.keyword.map(k => (
              <span key={k} className="keyword-badge">{k}</span>
            ))}
          </div>
          <div className="flow-stats">
            {flow.answers.length} responses configured
          </div>
          {flow.sensitive && (
            <span className="sensitive-badge">🔒 Sensitive</span>
          )}
        </div>
      ))}
    </div>
  );
}
```

```typescript Analyze Flow Complexity
import { getBotFlows } from './services/api';
import type { Flow } from './types';

async function analyzeFlowComplexity(builderBotId: string) {
  const flows = await getBotFlows(builderBotId);
  
  const analysis = flows.map(flow => {
    const hasAI = flow.answers.some(a => a.plugins.openai !== null);
    const hasMedia = flow.answers.some(a => a.options.media !== null);
    const hasGoto = flow.answers.some(a => a.options.gotoFlow !== null);
    const totalAnswers = flow.answers.length;
    
    return {
      name: flow.label,
      keywords: flow.keyword.length,
      answers: totalAnswers,
      features: {
        aiPowered: hasAI,
        includesMedia: hasMedia,
        hasFlowTransitions: hasGoto,
        transcribesAudio: flow.options.transcribeAudio,
        interpretsImages: flow.options.interpretImage
      },
      complexity: totalAnswers > 5 ? 'High' : totalAnswers > 2 ? 'Medium' : 'Low'
    };
  });
  
  console.log('Flow Analysis:', analysis);
  return analysis;
}
```

```typescript Map Flow Transitions
import { getBotFlows } from './services/api';
import type { Flow } from './types';

async function mapFlowTransitions(builderBotId: string) {
  const flows = await getBotFlows(builderBotId);
  
  // Build flow map
  const flowMap = new Map<string, Flow>();
  flows.forEach(flow => flowMap.set(flow.id, flow));
  
  // Find transitions
  const transitions: Array<{
    from: string;
    to: string;
    answerMessage: string;
  }> = [];
  
  flows.forEach(flow => {
    flow.answers.forEach(answer => {
      if (answer.options.gotoFlow) {
        const targetFlow = flowMap.get(answer.options.gotoFlow);
        transitions.push({
          from: flow.label,
          to: targetFlow?.label || 'Unknown',
          answerMessage: answer.message.substring(0, 50) + '...'
        });
      }
    });
  });
  
  return transitions;
}
```

## Common Use Cases

### Display Flow Summary

```typescript
async function displayFlowSummary(builderBotId: string) {
  const flows = await getBotFlows(builderBotId);
  
  console.log('=== Conversation Flows ===\n');
  
  flows.forEach((flow, index) => {
    console.log(`${index + 1}. ${flow.label}`);
    console.log(`   Triggers: ${flow.keyword.join(', ') || 'None'}`);
    console.log(`   Responses: ${flow.answers.length}`);
    console.log(`   AI-Powered: ${flow.answers.some(a => a.plugins.openai) ? 'Yes' : 'No'}`);
    console.log('');
  });
}
```

### Find AI-Powered Flows

```typescript
async function getAIPoweredFlows(builderBotId: string) {
  const flows = await getBotFlows(builderBotId);
  
  return flows.filter(flow =>
    flow.answers.some(answer => answer.plugins.openai !== null)
  );
}
```

### Extract All Keywords

```typescript
async function getAllKeywords(builderBotId: string): Promise<string[]> {
  const flows = await getBotFlows(builderBotId);
  
  const allKeywords = flows.flatMap(flow => flow.keyword);
  const uniqueKeywords = [...new Set(allKeywords)];
  
  return uniqueKeywords.sort();
}
```

## Error Handling

```typescript
import { getBotFlows } from './services/api';

async function safeGetFlows(builderBotId: string) {
  try {
    return await getBotFlows(builderBotId);
  } catch (error) {
    if (error instanceof Error) {
      if (error.message.includes('not found')) {
        console.error('BuilderBot instance not found');
        return [];
      } else if (error.message.includes('Failed to fetch')) {
        console.error('Network error loading flows');
        throw new Error('Cannot connect to server. Please check your connection.');
      }
    }
    throw error;
  }
}
```

## Important Notes

<Note>
The `builderBotId` parameter is the BuilderBot instance ID (`id_builderBot`), not the GestorPro bot ID (`id`).
</Note>

<Warning>
Flows with `deletedAt !== null` are soft-deleted and should typically be filtered out in your UI.
</Warning>

<Note>
The `answers` array contains the complete conversation logic, including OpenAI configurations, media attachments, and flow transitions.
</Note>

## Best Practices

1. **Cache flows** - They don't change frequently, so cache them locally
2. **Filter deleted flows** - Check `deletedAt` field
3. **Handle empty arrays** - Some bots may have no flows configured
4. **Parse OpenAI settings** - Extract AI configurations for analysis
5. **Map flow transitions** - Understand conversation paths
6. **Monitor complexity** - Track number of flows and answers per bot
