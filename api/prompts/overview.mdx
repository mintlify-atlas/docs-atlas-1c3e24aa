## Overview

The Prompts API allows you to manage AI prompts and their responses associated with projects in GestorPro. This is useful for storing AI-generated content, conversation history, or template prompts for project-related tasks.

## Prompt Interface

The `Prompt` type represents a prompt-response pair in the system:

```typescript
interface Prompt {
  id: string;              // Unique identifier
  projectId: string;       // Associated project ID
  prompt: string;          // The prompt text
  response: string;        // The response text
  createdAt: string;       // ISO timestamp
  isPinned?: boolean;      // Optional pinned status
}
```

### Add Prompt

Creates a new prompt for a project. The `id` and `createdAt` fields are automatically generated.

```typescript
const addPrompt = async (promptData: Omit<Prompt, 'id' | 'createdAt'>): Promise<Prompt>
```

<ParamField path="promptData" type="object" required>
  The prompt data to create (excluding auto-generated fields)
  
  <Expandable title="promptData properties">
    <ParamField path="projectId" type="string" required>
      The ID of the project this prompt belongs to
    </ParamField>
    
    <ParamField path="prompt" type="string" required>
      The prompt text
    </ParamField>
    
    <ParamField path="response" type="string" required>
      The response text
    </ParamField>
    
    <ParamField path="isPinned" type="boolean" optional>
      Whether to pin the prompt (defaults to false)
    </ParamField>
  </Expandable>
</ParamField>

<ResponseField name="prompt" type="Prompt">
  The created prompt object with generated `id` and `createdAt` fields
</ResponseField>

#### Example

```typescript
import { addPrompt } from './services/api';

const newPrompt = await addPrompt({
  projectId: 'proj_123',
  prompt: 'Create a technical specification for the authentication module',
  response: 'The authentication module will use JWT tokens...',
  isPinned: false
});

console.log(newPrompt);

//   response: "The authentication module will use JWT tokens...",
```

### Update Prompt

```typescript
const updatePrompt = async (promptData: Prompt): Promise<Prompt>
```

<ParamField path="promptData" type="Prompt" required>

    </ParamField>
    
    <ParamField path="projectId" type="string" required>

    </ParamField>
    
    <ParamField path="prompt" type="string" required>

    </ParamField>
    
    <ParamField path="response" type="string" required>

    <ParamField path="createdAt" type="string" required>

<ResponseField name="prompt" type="Prompt">

<Accordion title="deletePrompt" icon="trash">

### Delete Prompt

Deletes a prompt by its ID.

```typescript
const deletePrompt = async (promptId: string): Promise<{ success: boolean }>
```

<ParamField path="promptId" type="string" required>
  The unique identifier of the prompt to delete
</ParamField>

<ResponseField name="response" type="object">
  Deletion confirmation object
  
  <Expandable title="response properties">
    <ResponseField name="success" type="boolean">
      `true` if the prompt was successfully deleted, `false` otherwise
    </ResponseField>
  </Expandable>
</ResponseField>

#### Example

```typescript
import { deletePrompt } from './services/api';

const result = await deletePrompt('prompt_456');
console.log(result);
// { success: true }

if (result.success) {
  console.log('Prompt deleted successfully');
}
```

## Complete Usage Example

<CodeGroup>

```typescript React Component
import { useState, useEffect } from 'react';
import { 
  getPromptsByProjectId, 
  addPrompt, 
  updatePrompt, 
  deletePrompt 
} from './services/api';
import type { Prompt } from './types';

function ProjectPrompts({ projectId }: { projectId: string }) {
  const [prompts, setPrompts] = useState<Prompt[]>([]);
  const [loading, setLoading] = useState(true);

  // Load prompts
  useEffect(() => {
    const loadPrompts = async () => {
      try {
        const data = await getPromptsByProjectId(projectId);
        setPrompts(data);
      } catch (error) {
        console.error('Failed to load prompts:', error);
      } finally {
        setLoading(false);
      }
    };
    loadPrompts();
  }, [projectId]);

  // Create prompt
  const handleAddPrompt = async (prompt: string, response: string) => {
    const newPrompt = await addPrompt({
      projectId,
      prompt,
      response,
      isPinned: false
    });
    setPrompts([...prompts, newPrompt]);
  };

  // Update prompt
  const handleUpdatePrompt = async (prompt: Prompt) => {
    const updated = await updatePrompt(prompt);
    setPrompts(prompts.map(p => p.id === updated.id ? updated : p));
  };

  // Delete prompt
  const handleDeletePrompt = async (promptId: string) => {
    const result = await deletePrompt(promptId);
    if (result.success) {
      setPrompts(prompts.filter(p => p.id !== promptId));
    }
  };

  // Toggle pin
  const handleTogglePin = async (prompt: Prompt) => {
    await handleUpdatePrompt({
      ...prompt,
      isPinned: !prompt.isPinned
    });
  };

  if (loading) return <div>Loading prompts...</div>;

  return (
    <div>
      {prompts.map(prompt => (
        <div key={prompt.id}>
          <div>
            <strong>Prompt:</strong> {prompt.prompt} {prompt.isPinned && '📌'}
          </div>
          <div>
            <strong>Response:</strong> {prompt.response}
          </div>
          <button onClick={() => handleTogglePin(prompt)}>Toggle Pin</button>
          <button onClick={() => handleDeletePrompt(prompt.id)}>Delete</button>
        </div>
      ))}
    </div>
  );
}
```

## Error Handling

All Prompts API methods may throw errors. Always wrap calls in try-catch blocks:

```typescript
try {
  const prompts = await getPromptsByProjectId(projectId);
} catch (error) {
  if (error instanceof Error) {
    console.error('Failed to fetch prompts:', error.message);
  }
}
```

### Common Errors

- **Authentication Error**: Session not found or invalid
- **Connection Error**: Failed to reach the backend script
- **Validation Error**: Invalid prompt data provided
- **Not Found**: Prompt or project ID does not exist

## Best Practices

### Organizing Prompts

Use the `isPinned` field to highlight important AI-generated content:

```typescript
// Pin critical AI insights
await addPrompt({
  projectId: 'proj_123',
  prompt: 'Summarize project requirements',
  response: 'The project requires...',
  isPinned: true
});

// Sort prompts with pinned first, then by date
const sortedPrompts = prompts.sort((a, b) => {
  if (a.isPinned && !b.isPinned) return -1;
  if (!a.isPinned && b.isPinned) return 1;
  return new Date(b.createdAt).getTime() - new Date(a.createdAt).getTime();
});
```

### Prompt Templates

Create reusable prompt templates for common tasks:

```typescript
const PROMPT_TEMPLATES = {
  riskAnalysis: 'Analyze the risks for this project: ',
  requirementsSummary: 'Summarize the requirements for: ',
  technicalSpec: 'Create a technical specification for: '
};

async function createFromTemplate(
  projectId: string,
  template: string,
  context: string,
  aiResponse: string
) {
  return addPrompt({
    projectId,
    prompt: template + context,
    response: aiResponse,
    isPinned: false
  });
}
```

### Search and Filter

Implement client-side search for prompts:

```typescript
function searchPrompts(prompts: Prompt[], query: string): Prompt[] {
  const lowerQuery = query.toLowerCase();
  return prompts.filter(
    p => p.prompt.toLowerCase().includes(lowerQuery) ||
         p.response.toLowerCase().includes(lowerQuery)
  );
}

const results = searchPrompts(prompts, 'authentication');
```

### Conversation History

Build conversation threads from prompts:

```typescript
interface Conversation {
  projectId: string;
  messages: Array<{ role: 'user' | 'assistant'; content: string; timestamp: string }>;
}

function buildConversation(prompts: Prompt[]): Conversation {
  const sorted = prompts.sort((a, b) => 
    new Date(a.createdAt).getTime() - new Date(b.createdAt).getTime()
  );
  
  return {
    projectId: prompts[0]?.projectId || '',
    messages: sorted.flatMap(p => [
      { role: 'user' as const, content: p.prompt, timestamp: p.createdAt },
      { role: 'assistant' as const, content: p.response, timestamp: p.createdAt }
    ])
  };
}
```
