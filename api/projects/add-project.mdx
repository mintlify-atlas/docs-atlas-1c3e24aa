## Method Signature

```typescript
export const addProject = async (
  projectData: Omit<Project, 'id' | 'clientName'>
): Promise<Project>
```

**Source:** `~/workspace/source/services/api.ts:124`

Creates a new project in the Google Sheets backend. The backend automatically generates the `id` and populates `clientName` based on the provided `clientId`.

## Implementation Details

The method internally calls:

```typescript
return callAppsScript('addProject', projectData);
```

This sends a request to the Google Apps Script backend with:
- Action: `'addProject'`
- Payload: The project data object
- Sheet ID: Extracted from `localStorage` user session

import { addProject, ProjectStatus } from './services/api';
import type { Project } from './types';

interface ProjectFormData {
  name: string;
  clientId: string;
  startDate: string;
  endDate: string;
  budget: string;
  status: ProjectStatus;
}

async function handleProjectSubmit(formData: ProjectFormData) {
  // Validate dates
  const start = new Date(formData.startDate);
  const end = new Date(formData.endDate);
  
  if (end <= start) {
    throw new Error('End date must be after start date');
  }
  
  // Validate budget
  const budget = parseFloat(formData.budget);
  if (isNaN(budget) || budget <= 0) {
    throw new Error('Budget must be a positive number');
  }
  
  // Create project
  const project = await addProject({
    name: formData.name.trim(),
    clientId: formData.clientId,
    startDate: formData.startDate,
    endDate: formData.endDate,
    budget: budget,
    status: formData.status
  });
  
  return project;
}

## Validation Considerations

<Note>
  The API does not perform extensive validation. Implement client-side validation before calling this method.
</Note>

### Recommended Validations

```typescript
function validateProjectData(data: Omit<Project, 'id' | 'clientName'>): string[] {
  const errors: string[] = [];
  
  // Name validation
  if (!data.name || data.name.trim().length === 0) {
    errors.push('Project name is required');
  }
  if (data.name.length > 100) {
    errors.push('Project name must be less than 100 characters');
  }
  
  // Client validation
  if (!data.clientId) {
    errors.push('Client ID is required');
  }
  
  // Date validation
  const startDate = new Date(data.startDate);
  const endDate = new Date(data.endDate);
  
  if (isNaN(startDate.getTime())) {
    errors.push('Invalid start date');
  }
  if (isNaN(endDate.getTime())) {
    errors.push('Invalid end date');
  }
  if (endDate <= startDate) {
    errors.push('End date must be after start date');
  }
  
  // Budget validation
  if (typeof data.budget !== 'number' || data.budget < 0) {
    errors.push('Budget must be a positive number');
  }
  
  // Status validation
  const validStatuses = Object.values(ProjectStatus);
  if (!validStatuses.includes(data.status)) {
    errors.push('Invalid project status');
  }
  
  return errors;
}
```
