## Method Signature

```typescript
export const updateProject = async (
  projectData: Omit<Project, 'clientName'>
): Promise<Project>
```

**Source:** `~/workspace/source/services/api.ts:128`

Updates an existing project in the Google Sheets backend. The `clientName` field is omitted from input as it's automatically populated by the backend based on the `clientId`.

## Implementation Details

The method internally calls:

```typescript
return callAppsScript('updateProject', projectData);
```

This sends a request to the Google Apps Script backend with:
- Action: `'updateProject'`
- Payload: The complete project data object including `id`
- Sheet ID: Extracted from `localStorage` user session

import { getProjects, updateProject, ProjectStatus } from './services/api';

async function markProjectAsCompleted(projectId: string) {
  try {
    // Fetch current project data
    const projects = await getProjects();
    const project = projects.find(p => p.id === projectId);
    
    if (!project) {
      throw new Error('Project not found');
    }
    
    // Update only the status
    const updated = await updateProject({
      ...project,
      status: ProjectStatus.Completed
    });
    
    console.log(`Project "${updated.name}" marked as completed`);
    return updated;
  } catch (error) {
    console.error('Failed to update project status:', error);
    throw error;
  }
}

## Validation Considerations

### Recommended Validations

```typescript
function validateProjectUpdate(data: Omit<Project, 'clientName'>): string[] {
  const errors: string[] = [];
  
  // ID validation
  if (!data.id) {
    errors.push('Project ID is required');
  }
  
  // Name validation
  if (!data.name || data.name.trim().length === 0) {
    errors.push('Project name is required');
  }
  
  // Date validation
  const startDate = new Date(data.startDate);
  const endDate = new Date(data.endDate);
  
  if (endDate <= startDate) {
    errors.push('End date must be after start date');
  }
  
  // Budget validation
  if (typeof data.budget !== 'number' || data.budget < 0) {
    errors.push('Budget must be a positive number');
  }
  
  return errors;
}
```
