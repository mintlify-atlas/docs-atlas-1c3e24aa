## Method Signature

```typescript
export const deleteProject = async (
  projectId: string
): Promise<{ success: boolean }>
```

**Source:** `~/workspace/source/services/api.ts:132`

Deletes a project from the Google Sheets backend. This operation permanently removes the project and cannot be undone.

<Note>
  This is a destructive operation. Consider implementing a confirmation dialog before calling this method.
</Note>

## Parameters

<ParamField path="projectId" type="string" required>
  The unique identifier of the project to delete.
  
  ```typescript
  projectId: 'project-123'
  ```
</ParamField>

## Return Type

<ResponseField name="Promise<{ success: boolean }>" type="object">
  Returns a promise that resolves to an object indicating the operation result.
  
  <Expandable title="Success Response">
    <ResponseField name="success" type="boolean">
      Indicates whether the deletion was successful. Will be `true` if the project was deleted.
      
      ```typescript
      { success: true }
      ```
    </ResponseField>
  </Expandable>
</ResponseField>

## Implementation Details

The method internally calls:

```typescript
return callAppsScript('deleteProject', projectId);
```

This sends a request to the Google Apps Script backend with:
- Action: `'deleteProject'`
- Payload: The project ID string
- Sheet ID: Extracted from `localStorage` user session

```typescript With Confirmation
import { deleteProject } from './services/api';

async function handleDeleteProject(projectId: string, projectName: string) {
  const confirmed = window.confirm(
    `Are you sure you want to delete "${projectName}"? This action cannot be undone.`
  );
  
  if (!confirmed) {
    return;
  }
  
  try {
    const result = await deleteProject(projectId);
    
    if (result.success) {
      alert('Project deleted successfully');
      // Refresh project list or redirect
      window.location.href = '/projects';
    }
  } catch (error) {
    console.error('Failed to delete project:', error);
    alert('Failed to delete project. Please try again.');
  }
}
```

```typescript With Error Handling
import { deleteProject } from './services/api';

async function safeDeleteProject(projectId: string): Promise<boolean> {
  try {
    const result = await deleteProject(projectId);
    return result.success;
  } catch (error) {
    if (error instanceof Error) {
      if (error.message.includes('not found')) {
        console.warn('Project already deleted or does not exist');
        return true; // Consider it success if already gone
      }
      
      if (error.message.includes('not authenticated')) {
        // Redirect to login
        window.location.href = '/login';
      }
      
      console.error('Delete error:', error.message);
    }
    return false;
  }
}
```

```typescript React Component
      </div>
    );
  }

  return (
    <button 
      onClick={() => setShowConfirm(true)}
      className="btn-danger"
```

```typescript Bulk Delete
import { deleteProject } from './services/api';

async function deleteMultipleProjects(
  projectIds: string[]
): Promise<{ succeeded: string[], failed: string[] }> {
  const succeeded: string[] = [];
  const failed: string[] = [];
  
  for (const id of projectIds) {
    try {
      const result = await deleteProject(id);
      
      if (result.success) {
        succeeded.push(id);
      } else {
        failed.push(id);
      }
    } catch (error) {
      console.error(`Failed to delete project ${id}:`, error);
      failed.push(id);
    }
  }
  
  console.log(`Deleted ${succeeded.length} projects, ${failed.length} failed`);
  return { succeeded, failed };
}

const projectsToDelete = ['project-1', 'project-2', 'project-3'];
```

## Important Considerations

<Note>
  Deleting a project may leave orphaned data in related entities (notes, links, prompts). Consider implementing cascade deletion or cleanup logic.
</Note>

### Related Data Cleanup

```typescript
import { 
  deleteProject, 
  getNotesByProjectId, 
  deleteNote,
  getLinksByProjectId,
  deleteLink,
  getPromptsByProjectId,
  deletePrompt
} from './services/api';

async function deleteProjectWithRelatedData(projectId: string) {
  try {
    // Delete related notes
    const notes = await getNotesByProjectId(projectId);
    for (const note of notes) {
      await deleteNote(note.id);
    }
    
    // Delete related links
    const links = await getLinksByProjectId(projectId);
    for (const link of links) {
      await deleteLink(link.id);
    }
    
    // Delete related prompts
    const prompts = await getPromptsByProjectId(projectId);
    for (const prompt of prompts) {
      await deletePrompt(prompt.id);
    }
    
    // Finally, delete the project
    const result = await deleteProject(projectId);
    
    console.log('Project and all related data deleted');
    return result;
  } catch (error) {
    console.error('Failed to delete project with related data:', error);
    throw error;
  }
}
```
